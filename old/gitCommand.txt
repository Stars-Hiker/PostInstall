
# Initialize a new Git repository
git init  # Creates a new Git repository in the current directory

# Add files to the staging area
git add <file>  # Stages a specific file
git add .       # Stages all files in the current directory and subdirectories

# Commit changes
git commit -m "message"  # Commits staged changes with a descriptive message

# Check the status of the working directory
git status  # Shows the status of tracked, staged, and untracked files

# View commit history
git log  # Displays the commit history in reverse chronological order

# Set up a remote repository
git remote add origin <URL>  # Links the local repository to a remote repository

# Update the URL of an existing remote repository
git remote set-url origin <new-URL>  # Changes the remote repository URL

# Remove an existing remote
git remote remove origin  # Deletes the remote named 'origin'

# Push changes to the remote repository
git push -u origin main  # Pushes the local 'main' branch to the remote 'origin'

# Pull changes from the remote repository
git pull origin main  # Fetches and merges changes from the remote 'main' branch

# Clone a repository
git clone <URL>  # Creates a local copy of the repository at the specified URL

# Create a new branch
git branch <branch-name>  # Creates a new branch but does not switch to it

# Switch to a branch
git checkout <branch-name>  # Switches to the specified branch (older syntax)
git switch <branch-name>    # Switches to the specified branch (modern syntax)

# Merge branches
git merge <branch-name>  # Merges the specified branch into the current branch

# Discard changes in a file
git checkout -- <file>  # Reverts a file to the last committed state

# Delete a branch
git branch -d <branch-name>  # Deletes a branch safely (checks if merged)
git branch -D <branch-name>  # Force deletes a branch

# Show changes
git diff  # Displays changes between the working directory and staging area
git diff --staged  # Shows changes between staging area and last commit

# Tagging a commit
git tag -a <tag-name> -m "message"  # Creates an annotated tag for a commit
git push origin <tag-name>  # Pushes the tag to the remote repository

# Configure user information
git config --global user.name "Your Name"  # Sets your Git username globally
git config --global user.email "you@example.com"  # Sets your email globally

# View remote repositories
git remote -v  # Lists remote repositories and their URLs

# Remove all local changes (CAUTION)
git reset --hard  # Resets the working directory to the last commit

# Stash changes
git stash  # Temporarily saves uncommitted changes
git stash apply  # Applies the most recent stashed changes

# Show help
git help <command>  # Displays help for a specific Git command
